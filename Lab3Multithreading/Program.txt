kernel void matrix_multiplication(global const float* a, global const float* b, global float* c, int wA, int wB)
{
	int tx = get_global_id(0); 
	int ty = get_global_id(1);

   float value = 0.0;
   for (int k = 0; k < wA; ++k)
   {
      float elementA = a[ty * wA + k];
      float elementB = b[k * wB + tx];
      value += elementA * elementB;
   }

   c[ty * wB + tx] = value;
}

kernel void matricesMul(global const float* in1, global const float* in2, global float* out, int COLSROWS, int COLS2)
{

    int r = get_global_id( 0 );
    int c = get_global_id( 1 );  

    for( int cr = 0; cr < COLSROWS; cr ++ )
        out[ r * COLS2 + c ] += in1[ r * COLSROWS + cr ] * in2[ cr + c * COLSROWS ];

}
